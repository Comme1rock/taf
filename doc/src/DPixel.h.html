  Double_t           fCommonMode;                   // the common mode value 
  Double_t           fPedestal;                     // pedestal of strip
  Int_t              fTimestamp;                    // timestamp associated, 2013/06/22

  Bool_t             fFound;                        // flag, that strip is found in hit

};
