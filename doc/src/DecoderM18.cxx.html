Bool_t DecoderM18::ReadNextInt() {

  if(!fFileInput){ printf("Error No Input File\n"); return 0; }
  
  fread(&fDataChar,1,4,fFileInput);
   
  if( feof(fFileInput) ) {
    printf("DecoderM18: No more data in file! -> STOPPING\n");
    return kFALSE;
  }
    
 return kTRUE;
}

Int_t DecoderM18::Get_Nevent() {
  
  if(!fFileInput){ printf("Error No Input File\n"); return -1; }

  Long_t position;
  position = ftell(fFileInput); //save current position 


  fseek(fFileInput,0,SEEK_SET); // go to beginning of the file


  fNEvent=0;
  while(!feof(fFileInput))
    {
      if(!ReadNextInt()){
	cout << "No 32 bit words to read!" << endl;
	return -1;
      }
      if(fDataChar == 0xfafafafa){fNEvent++;}
    }


  fseek(fFileInput,position,SEEK_SET); //go back to initial position


  return fNEvent;



}

Bool_t DecoderM18::ReadCDH() {
    
  do{
    if(!ReadNextInt()) return kFALSE;
  }
  while(fDataChar != 0xfafafafa);

  NewEvent();

  Int_t NwordJump=0;

  fcountEv++;
 
  if(!ReadNextInt()) return kFALSE;
  fEvCounterCDH = fDataChar;
  if(fcountEv!=fEvCounterCDH) 
    cout << "N ev mismatch!!! counter: " << fcountEv << " read from CDH " << fEvCounterCDH << endl;
  if(!ReadNextInt()) return kFALSE; //Dummy info (stop pointer)

  if(!ReadNextInt()) return kFALSE; //Dummy info (wrap around bit)

  if(!ReadNextInt()) return kFALSE; 

  NwordJump=fDataChar; //(OMKDtransition) # of words to skip

  for(Int_t i=0; i<NwordJump; i++) if(!ReadNextInt()) return kFALSE;

   
    
return kTRUE;
    
}
// #############################################################################

void DecoderM18::NewEvent() {

  fEvCounterCDH=0;
  fIndex.clear();
  fAmp.clear();


}



// #############################################################################

Bool_t DecoderM18::ReadData() {

  if(!ReadCDH()) return kFALSE;
  Int_t Amp=0;
  
  if (fDebugLevel) printf("DecoderM18::ReadData reading event %d\n", fEvCounterCDH);

  while(true){
    if(!ReadNextInt()) return kFALSE;
    if(fDataChar != 0xfafafafa){
      fIndex.push_back(CalculateIndex());
      Amp = (fDataChar & 0xffff0000) >> 16;
      fAmp.push_back(Amp-kToBeSutractedFromAmp);
      if (fDebugLevel) printf("  event %d (real #=%d) completed with %d pixels found\n", fEvCounterCDH, fcountEv, fIndex.size());
      //cout << hex << fDataChar << endl;

    } 
    else {
      fseek(fFileInput,-4,SEEK_CUR);
      return kTRUE;
    }
  }
 
    

    
return kTRUE;
    
}


// #############################################################################

Int_t DecoderM18::CalculateIndex() {

  Int_t row ,col;



  row = fDataChar & 0x000000FF;
  col = (fDataChar & 0x0000FF00) >> 8;

  //  cout << "row " << row << " col " << col << endl;



  Int_t index =row *kRowPerChip +col;
  if(index>=kShift) index = index -kShift;
  else index = kRowPerChip*kColPerChip + index - kShift;    
   
    
  return index;
    
}
